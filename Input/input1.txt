SENCER fact;

FUNCIO FACTORIAL(SENCER n): SENCER {
    SENCER resultat;

    resultat = 1;
    // sample comment

    resultat = resultat * n;
    n = n - 1;

    MENTRE n != 0 FER
        resultat = resultat * n;
        n = n - 1;
    FIMENTRE;

    RETORNAR resultat;
};

FUNCIO PRINCIPAL(): SENCER{

    SENCER num;
    num = 1;

    ESCRIURE("Valor inicial de num: ", num);
// sample comment

    SI num > 0 LLAVORS
        ESCRIURE("El número és positiu");
    FISI;

    SI num < 0 LLAVORS
        ESCRIURE("El número és zero o negatiu");
    FISI;


    MENTRE num > 0 FER
        ESCRIURE("Valor actual de num: ", num);
        num = num - 1;
    FIMENTRE;


    fact = FACTORIAL(5);
    ESCRIURE("Factorial de 5: ", fact);
};



SENCER a;
INICI
    a = 1;
    PRINCIPAL();
FI
